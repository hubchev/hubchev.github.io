# Alternative with pipe operator:
carsSub <- cars %>%
  filter(cyl == 8) %>%
  select(wt, qsec, hp, car)

# 16

# 17
carsSub <- carsSub %>%

# Alternatively you can put everything into one pipe:
carsSub2 <- cars %>%
  filter(cyl == 8) %>%
  select(wt, qsec, hp, car) %>% 
  arrange(carsSub, wt) %>% 
  mutate(wt2 = wt^2)
